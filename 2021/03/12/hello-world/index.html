import numpy as np
import matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = ['SimHei']  # 用来正常显示中文标签
plt.rcParams['axes.unicode_minus'] = False  # 用来正常显示负号#有中文出现的情况，需要u'内容'
# 训练集
y = np.array([-890, -1411, -1560, -2220, -2091, -2878, -
              3537, -3268, -3920, -4163, -5471, -5157])
x = np.array([1, 2, 2, 3, 3, 4, 5, 6, 6, 6, 8, 10])
# 参数设置
eta = 0.001
train_circles = 5000
w = -700
b = 0
n = x.size
Loss = np.ones(train_circles)
for step in range(train_circles):
    Loss[step] = sum((y-w*x-b)*(y-w*x-b))/n
    w = w+eta*2/n*sum((y-w*x-b)*x)
    b = b+eta*2/n*sum(y-w*x-b)

# Result=[w,b,Loss[-1]]
print('w = ' + str(w)+ ', ' + 'b = '+str(b) + ', ' + 'Loss = '+str(Loss[-1]))
#绘制图一
n = np.linspace(1, train_circles, train_circles)
plt.figure(1, figsize=(7,13))
plt.subplot(211)
plt.plot(n, Loss,color='black',linewidth='1.1')
plt.xlabel('迭代次数')
plt.ylabel('拟合均方误差')
plt.title('迭代次数与拟合误差关系曲线')
#plt.show()
#绘制图2
n = np.linspace(1, 10, 10)
#plt.figure()
plt.subplot(212)
plt.plot(n, n*w+b, color='green',linewidth='1.1')
plt.scatter(x, y,marker='o', c='m',linewidths='0.7')
plt.xlabel('碳原子数')
plt.ylabel('燃烧热')
plt.title('碳原子数与燃烧热的拟合直线')
plt.show()
